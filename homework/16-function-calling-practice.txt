def get_birzai_wiki():
    response = requests.get(
        "https://en.wikipedia.org/w/api.php",
        params={
            "action": "query",
            "format": "json",
            "prop": "extracts",
            "exintro": True,
            "explaintext": True,
            "titles": "Bir≈æai"
        }
    )
    data = response.json()
    page = next(iter(data["query"]["pages"].values()))
    return page["extract"]
